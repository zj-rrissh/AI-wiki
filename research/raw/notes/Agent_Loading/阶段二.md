对话类 可自定义人格，记录token

```python
import os
import tiktoken
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

class Talk:
    def __init__(self, model="deepseek-chat",system_prompt="你是一个有用的助手"):
        self.client = OpenAI(api_key = os.getenv("API_KEY"),
                            base_url="https://api.deepseek.com")
        self.model = model
        self.enc = tiktoken.encoding_for_model(model)
        self.messages = [{"role": "system", "content": system_prompt}]
        self.total_tokens = 0

    def set_persona(self, persona):
        self.messages[0] = {
            "role": "system",
            "content": persona
        }
    
    def add_user_message(self, message):
        self.messages.append({
            "role": "user",
            "content": message
        })

    def chat(self,temperature=0.7, top_p=1.0, presence_penalty=0.0):
        response = self.client.chat.completions.create(
            model = self.model,
            messages = self.messages, # type: ignore
            temperature = temperature,
            top_p = top_p,
            presence_penalty = presence_penalty,
        )

        reply = response.choices[0].message.content
        if reply is None:
            reply = ""

        self.messages.append({
            "role": "assistant",
            "content": reply # type: ignore
        })

        self.total_tokens += self._count_tokens(self.messages)

        return reply
    
    def _count_tokens(self, messages):
        tokens = 0
        for message in messages:
           tokens += len(self.enc.encode(message["content"])) + 4  # 每条消息固定4个tokens
        return tokens + 2  # 开始和结束的tokens
    
    def get_token_usage(self):
        return self.total_tokens
```





**tool_call.py**

```python
import requests
import json
import re

tools = [
    {
        "name": "get_weather",
        "description": "获取指定城市的天气信息",
        "parameters": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "要查询天气的城市名称"
                }
            },
            "required": ["city"]
        }
    },
    {
        "name": "calculator",
        "description": "执行基本的数学计算",
        "parameters": {
            "type": "object",
            "properties": {
                "a":{"type":"number"},
                "b":{"type":"number"},
                "op":{
                    "type":"string",
                    "enum":["+","-","*","/"]
                }
            },
            "required": ["a", "b", "op"]
        }
    }
]

def build_prompt(user_input,tools):
    return f"""
你是一个智能助手，可以调用工具完成任务。

工具列表:
{tools}

规则:
1. 如果需要调用工具，必须输出JSON:
{{
    "tool": "工具名",
    "args": {{参数}}
}}
2.不需要输出解释
3.如果不需要调用工具，直接输出:
{{
    "tool": null,
    "answer": "回答内容"
}}

用户输入: 
{user_input}
"""

def call_llm(prompt):
    response = requests.post(
        "http://localhost:11434/api/generate",
        json={
            "model": "qwen2.5:7b",
            "prompt": prompt,
            "stream": False
        }
    )

    return response.json()["response"]

def parse_response(resp):
    try:
        return json.loads(resp)
    except:
        match = re.search(r'\{.*\}', resp, re.S)
        if match:
            try:
                return json.loads(match.group())
            except:
                pass
        return {"tool": None, "answer": resp}
    
def get_weather(city):
    return f"{city}天气晴，25度"

def calculator(a,b,op):
    if op == "+": return a + b
    if op == "-": return a - b
    if op == "*": return a * b
    if op == "/": return a / b

tool_map = {
    "get_weather": get_weather,
    "calculator": calculator
}

def execute_tool(tool_name,args):
    if tool_name not in tool_map:
        return "工具不存在"
    return tool_map[tool_name](**args)

def agent(user_input):
    prompt = build_prompt(user_input,tools)

    resp = call_llm(prompt)
    data = parse_response(resp)

    if data["tool"]:
        result = execute_tool(data["tool"],data["args"])

        final_prompt = f"""
用户问题: {user_input}
工具返回结果: {result}

请生成最终回答
"""
        return call_llm(final_prompt)
    else:
        return data["answer"]
    

def chat():
    print("Agent 启动")
    while True:
        user_input = input("user: ")
        if user_input.lower() in ["exit","quit"]:
            break

        result = agent(user_input)
        print("AI: ", result)

if __name__ == "__main__":
    chat()
```
